Using Model-Based Forms:

Angular provides another approach, known as model-based forms, in which the details of the form and its validation are defined in code rather than a template.
The model-based forms feature is defined in a module called ReactiveFormsModule, which is defined in the @angular/forms JavaScript module.

and the FormGroup class is used to manage a form element and its contents.

src\app\template.html-------------------------------------------------
  <form class="m-2" novalidate [formGroup]="formGroup2" (ngSubmit)="submitForm()">  //把model和form连在一起。

  <div class="form-group" *ngFor="let control of formGroup2.productControls">   //Generating the Elements from the Model
      <label>{{control.label}}</label>
      <input class="form-control" name="{{control.modelProperty}}" formControlName="{{control.modelProperty}}" />
      <ul class="text-danger list-unstyled" *ngIf="(formSubmitted || control.dirty) && control.invalid">
          <li *ngFor="let error of control.getValidationMessages()">
              {{error}}
          </li>
      </ul>
  </div>

    formGroup: ProductFormGroup = new ProductFormGroup();

    submitForm() {
        Object.keys(this.formGroup.controls)
            .forEach(c => this.newProduct[c] = this.formGroup.controls[c].value);   //克隆element
        this.formSubmitted = true;
        if (this.formGroup.valid) {   //触发
            this.addProduct(this.newProduct);
            this.newProduct = new Product();
            this.formGroup.reset();
            this.formSubmitted = false;
        }
    }
