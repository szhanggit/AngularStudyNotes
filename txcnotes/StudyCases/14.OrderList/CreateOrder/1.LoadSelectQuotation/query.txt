  getLanguages(): Observable<Dictionary[]> {
    const url = `${this._getURL()}api/GraphQL/Query`;

      query: `
      query { 
        dictionaries (where: { category: { eq: "CultureCode"}}) { 
          dictionaryId,
          name,
          displayName

      }`,

      this.http.post(url, body) as Observable<BaseResponse>
