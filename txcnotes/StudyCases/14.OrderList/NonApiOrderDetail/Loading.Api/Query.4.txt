  getExpirationPolicies() {
    const url = `${this.getURL()}api/GraphQL/Query`;
    const body = {
      query: `{
        expirationPolicies {
          id
          name
        }
      }`,
    };
    return (this.http.post(url, body) as Observable<BaseResponse>).pipe(
      map((response: BaseResponse) => {
        if (response.success) {
          return JSON.parse(response.data)?.expirationPolicies;
        }
      })
    );
  }
