    const skipClause = skip ? `, skip: ${skip}` : '';

    const body = { query:
      `query{
        monoAcceptanceLoopByMerchantId(merchantId: ${merchantId}
          order: [{isDefault: DESC}, { createdOn: DESC }]

          ${takeClause}
          )
          {
            totalCount
              items {
                acceptanceLoopId
                code
                description
                isDefault
                shopCountAvailableInAL
                status
                createdBy
                createdOn
              }
          }
        }`
